package BaldwinBeans;
import BaldwinBeans.*;
import java.beans.*;

public class jspBeanTest002BeanInfo 
                    extends SimpleBeanInfo{

  public PropertyDescriptor[] 
                  getPropertyDescriptors(){
    try{
      PropertyDescriptor propDesc = 
                    new PropertyDescriptor(
                            "size",
                            jspBeanTest002.class,
                            "xgetSize",
                            "xsetSize");
      PropertyDescriptor[] propDescArray = 
                                {propDesc};
      return propDescArray;
    }catch(IntrospectionException e){
      return null;
    }//end catch
  }//end getPropertyDescriptors()

}//end class jspBeanTest002BeanInfo
